# Перебор последовательностей с помощью for

# Прямой перебор элементов списка
fruits = ["apple", "banana", "cherry"]
for fruit in fruits:
    print("Фрукт:", fruit)
>>> Фрукт: apple
>>> Фрукт: banana
>>> Фрукт: cherry

# Перебор индексов с помощью range
# Используем функцию range для 
# перебора индексов
for i in range(len(fruits)):
    print(f"Фрукт {i + 1}: {fruits[i]}")
>>> Фрукт 1: apple
>>> Фрукт 2: banana
>>> Фрукт 3: cherry

# Использование функции enumerate
# Возвращает и индекс, и элемент
for index, fruit in enumerate(fruits):
    print(f"Фрукт {index + 1}: {fruit}")
>>> Фрукт 1: apple
>>> Фрукт 2: banana
>>> Фрукт 3: cherry

# Перебор последовательностей с помощью 
# for может быть осуществлен тремя 
# способами: прямым перебором элементов,
# перебором индексов с помощью range и 
# использованием enumerate, который позволяет
# одновременно получать и индекс, и значение.
