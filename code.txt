# Циклы позволяют выполнять блок кода многократно
# это полезно для повторяющихся задач

# Пример задачи без использования цикла
# Допустим, нужно напечатать числа от 1 до 5

print(1)
print(2)
print(3)
print(4)
print(5)
>>> 1
>>> 2
>>> 3
>>> 4
>>> 5

# Решение той же задачи с использованием цикла
# Цикл for

for i in range(1, 6):
    print(i)
>>> 1
>>> 2
>>> 3
>>> 4
>>> 5

# Цикл while также может быть использован 
# для этой задачи
i = 1
while i <= 5:
    print(i)
    i += 1
>>> 1
>>> 2
>>> 3
>>> 4
>>> 5

# Циклы позволяют упростить код 
# и сделать его более гибким.
