# Типы данных 
# Зачем нужны? 
# С разной информацией работаем 
# по-разному. 

# Числа нужны для вычислений, например сложение
num1 = 10  # целое число
num2 = 5.5  # число с плавающей точкой
result = num1 + num2  # складываем числа

print("Результат сложения чисел:", result)
# Результат сложения чисел: 15.5

# Текст(строки) можно соединять
#, получать фрагменты, преобразовывать
str1 = "Hello, "  # строка текста
str2 = "world!"  # еще одна строка текста
greeting = str1 + str2  # объединение строк

print("Результат сложения строк:", greeting)
# Результат сложения строк: Hello, world!

# Основные типы данных в Python:
integer_var = 42  # целые числа
float_var = 3.14  # числа с плавающей точкой
string_var = "Python"  # строки текста
bool_var = True  # логические значения (True/False)

# Важно понимать, что разные типы данных
# предназначены для различных целей и операций
